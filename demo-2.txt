function rotate_arr(arr,k){
 
 for(var a=0;a<k;a++){
     var end = arr[0]
     for(var i=0;i<arr.length-1;i++){
      arr[i] = arr[i+1]   
     }
     arr[arr.length-1] = end
} 

return arr;
}
